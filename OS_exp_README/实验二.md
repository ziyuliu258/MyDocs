### 代码展示
#### 流程图实现
```c
#include <stdio.h>
#include <unistd.h>
#include <sys/wait.h>
#include <stdlib.h>
#include <signal.h>
int flag = 0;
void inter_handler()
{
    flag = 1;
    printf("Signal received!!\n");
}
void waiting()
{
    while (!flag); // 忙等
}
int main()
{
    // TODO: 五秒之后或接收到两个信号
    signal(SIGINT, inter_handler);
    signal(SIGQUIT, inter_handler);
    signal(SIGALRM, inter_handler); // 5秒倒计时
    pid_t pid1 = -1, pid2 = -1;
    while (pid1 == -1)
        pid1 = fork(); // 创建子进程 1
    if (pid1 > 0)
    {
        while (pid2 == -1)
            pid2 = fork(); // 创建子进程 2
        if (pid2 > 0)
        {
            // TODO: 父进程
            // sleep(1); 这句加了很可能导致父进程错过手动信号，从而无法按照预期工作。
            flag = 0;
            alarm(5); // 设置5秒闹钟
            waiting();
            /*
            waiting() 会利用flag检测用 signal() 注册的所有信号，相当于默认的pause()。
            这里的原理如下：
            1. 父进程设置一个5秒的闹钟，并调用waiting()进入忙等状态，等待信号。
            2. 如果在5秒内按下中断键（Ctrl+C），会触发SIGINT信号，调用inter_handler函数，将flag设置为1，waiting()返回。
            3. 如果5秒内没有按下中断键，闹钟时间到达，触发SIGALRM信号，调用inter_handler函数，将flag设置为1，waiting()返回。
            4. 无论是哪种情况，父进程都会在waiting()返回后继续执行，取消闹钟，并向两个子进程发送信号
            */
            // 向两个子进程发送信号
            if (flag == 1)
            {
                alarm(0); // 取消闹钟，但不会清空已经发送的SIGALRM信号
                printf("\nParent sending signals to children...\n");
                flag = 0; // 还原flag
                kill(pid1, SIGUSR1);
                kill(pid2, SIGUSR2);
            }

            // 等待两个子进程终止
            waitpid(pid1, NULL, 0);
            waitpid(pid2, NULL, 0);

            printf("\nParent process is killed!!\n");
        }
        else
        {
            // TODO: 子进程 2
            signal(SIGINT, SIG_IGN); // 恢复默认行为
            signal(SIGQUIT, SIG_IGN);
            signal(SIGALRM, SIG_IGN);
            flag = 0;
            signal(SIGUSR2, inter_handler);
            waiting();
            printf("Child process2 is killed by parent!!\n");
            return 0;
        }
    }
    else
    {
        // TODO:子进程
        signal(SIGINT, SIG_IGN); // 恢复默认行为
        signal(SIGQUIT, SIG_IGN);
        signal(SIGALRM, SIG_IGN);
        flag = 0;
        signal(SIGUSR1, inter_handler);
        waiting();
        printf("Child process1 is killed by parent!!\n");
        return 0;
    }
    return 0;
}
```

#### 运行结果猜测

**猜测的运行结果**
我最初认为程序的运行流程应该是：
1. **程序启动**后，父进程创建两个子进程
2. **两个子进程立即进入等待状态**，通过 `waiting()` 函数忙等，等待父进程发送 `SIGUSR1` 和 `SIGUSR2` 信号
3. **父进程设置 5 秒闹钟**并进入等待状态
4. **如果 5 秒内按下 Ctrl+C**：
    - 触发 `SIGINT` 信号
    - 父进程从 `waiting()` 返回
    - 立即向两个子进程发送 `SIGUSR1` 和 `SIGUSR2` 信号
    - 子进程收到信号后打印消息并退出
    - 父进程等待子进程结束后打印消息并退出
5. **如果 5 秒后没有手动中断**：
    - 闹钟到达，触发 `SIGALRM` 信号
    - 执行相同的流程：父进程发送信号 → 子进程退出 → 父进程退出

#### **两种方式的差别对比**

| 对比项      | 手动中断（SIGINT）               | 自动触发（SIGALRM） |
| -------- | -------------------------- | ------------- |
| **触发方式** | 按下`^C`或者`^\`               | 系统闹钟到达        |
| **响应时间** | 立即（< 0.1秒）                 | 固定5秒          |
| **灵活性**  | 高（可随时触发）                   | 低（必须等待5秒）     |
| **用途场景** | 测试、调试、提前终止                 | 超时保护、自动执行     |
| **执行流程** | 完全一致                       | 完全一致          |
#### 中断方式更改

## 2.2 管道通信
### 程序流程图：

### 代码展示
#### 示例代码补全
```c
/*管道通信实验程序残缺版 */
#include <unistd.h>
#include <signal.h>
#include <stdio.h>
#include <sys/wait.h>
#include <stdlib.h> //for exit()
int pid1, pid2;
// 定义两个进程变量
int main()
{
    int fd[2];
    char InPipe[1001]; // 定义读缓冲区
    char c1 = '1', c2 = '2';
    pipe(fd); // 创建管道
    while ((pid1 = fork()) == -1); // 如果进程 1 创建不成功,则空循环
    if (pid1 == 0)
    {
        close(fd[0]); 
        // 如果子进程 1 创建成功,pid1 为进程号
        lockf(fd[1], 1, 0);// 锁定管道
        for (int i = 0; i < 2000; i++) write(fd[1], &c1, 1); // 分 2000 次每次向管道写入字符’1’
        sleep(2); // 等待读进程读出数据
        lockf(fd[1], 0, 0);
        exit(0); // 结束进程 1
    }
    else
    {
        while ((pid2 = fork()) == -1); // 若进程 2 创建不成功,则空循环
        if (pid2 == 0)
        {
            close(fd[0]); 
            lockf(fd[1], 1, 0);
            for (int i = 0; i < 2000; i++) write(fd[1], &c2, 1);
            //  分 2000 次每次向管道写入字符’2’
            sleep(2); // 同步，等待读进程
            lockf(fd[1], 0, 0);
            exit(0);
        }
        else // 父进程执行部分
        {
            close(fd[1]);
            wait(0); //  等待子进程 1 结束
            wait(0); // 等待子进程 2 结束
            int current = 0;
            int buffer_size = 1000;
            while (current < 4000)
            {
                size_t to_read = (4000 - current < buffer_size) ? (4000 - current) : buffer_size;
                ssize_t n = read(fd[0], InPipe, to_read);

                if (n <= 0)
                    break; // EOF 或错误
                InPipe[n] = '\0'; // 添加结束符
                printf("%s", InPipe);
                current += n;
            }
            printf("\n");
            // 显示读出的数据
            close(fd[0]);
            exit(0);
            // 父进程结束
        }
    }
}
```
